Quick examples

Hello World

include Carbuncle

class Example < Game
  def load
    @text = Text.new
    @text.value = 'Hello World'
    self << @text # 'add_child @text' also works
  end
end

Drawing an Image

include Carbuncle

class Example < Game
  def load
    texture = Texture.new('img/carbuncle.png')
    @sprite = Sprite.new(texture)
    @sprite.position.set(center)
    @sprite.pivot.set(0.5, 0.5)
  end

  def draw
    @sprite.draw
  end

  def center
    Point.new(
      screen.width / 2,
      screen.height / 2
    )
  end
end

Playing music and sound

include Carbuncle

class Example < Game
  def load
    Audio.music_play 'bgm/awesome.ogg'
    @sound = Sound.new('sfx/click.wav')
    @music.play
  end

  def update(dt)
    super(dt)
    @sound.play if Keyboard.press?(:space)
  end
end
