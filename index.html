Drawing an Image

1include Carbuncle
2
3class Example < Game
4  def load
5    texture = Texture.new('img/carbuncle.png')
6    @sprite = Sprite.new(texture)
7    @sprite.position.set(center)
8    @sprite.pivot.set(0.5, 0.5)
9  end
10
11  def draw
12    @sprite.draw
13  end
14
15  def center
16    Point.new(
17      screen.width / 2,
18      screen.height / 2
19    )
20  end
21end

Playing music and sound

1include Carbuncle
2
3class Example < Game
4  def load
5    @music = Music.new('bgm/awesome.ogg')
6    @sound = Sound.new('sfx/click.wav')
7    @music.play
8  end
9
10  def update(dt)
11    @sound.play if Keyboard.press?(:space)
12  end
13end
