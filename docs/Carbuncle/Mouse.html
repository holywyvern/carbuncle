Overview

This class handles mouse input. On touch screens, the first touch you do at the screen is interpreted as the mouse.

static mrb_value
mrb_mouse_set_position(mrb_state *mrb, mrb_value self)
{
  mrb_value obj;
  mrb_int x, y, argc;
  argc = mrb_get_args(mrb, "o|i", &obj, &y);
  if (argc == 1)
  {
    Vector2 *point = mrb_carbuncle_get_point(mrb, obj);
    x = point->x;
    y = point->y;
  }
  else
  {
    x = mrb_fixnum(mrb_to_int(mrb, obj));
  }
  SetMousePosition(x, y);
  return obj;
}
