Quick examples

Hello world

include Carbuncle

class Example < Game
  def load
    Font.default_name = 'fonts/some-font.ttf'
    @text = Text.new
    @text.value = 'Hello World'
  end

  def draw
    @text.draw
  end
end

Drawing an image

include Carbuncle

class Example < Game
  def load
    texture = Texture.new('img/carbuncle.png')
    @sprite = Sprite.new(texture)
    @sprite.position.set(screen.width / 2, screen.height / 2)
    @sprite.pivot.set(0.5, 0.5)
  end

  def draw
    @sprite.draw
  end
end

Playing Music & Sound

include Carbuncle

class Example < Game
  def load
    @music = Music.new('bgm/awesome.ogg')
    @sound = Sound.new('sfx/click.wav')
    @music.play
  end

  def update(dt)
    @sound.play if Keyboard.press?(:space)
  end
end
